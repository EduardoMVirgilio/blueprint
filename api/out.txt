```
model User {
  id        Int      @id @default(autoincrement())
  name      String
  age       Int
  isStudent Boolean
  address   Address
}

model Address {
  id       Int      @id @default(autoincrement())
  street   String
  city     String
  zip      Int
  user     User     @relation(fields: [userId], references: [id])
  userId   Int
}
```
